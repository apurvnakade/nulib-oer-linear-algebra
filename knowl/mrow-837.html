\begin{align*} p_{A'}(t)\amp =\det(tI-A')\\ \amp =\det(tI-P^{-1}AP) \amp (A'=P^{-1}AP)\\ \amp = \det(P^{-1}tIP-P^{-1}AP) \amp (\text{algebra}) \\ \amp = \det(P^{-1}(tI-A)P) \amp (\text{left/right dist.}) \\ \amp = \det(P^{-1})\det(tI-A)\det(P) \amp (\knowl{./knowl/th_det_mult.html}{\text{2.5.26}}) \\ \amp = (\det(P))^{-1}\det(P)\det(tI-A) \\ \amp = \det(tI-A)=p_A(t) \end{align*}
